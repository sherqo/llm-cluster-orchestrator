C.3 Worker Selection Strategy

File:

master/lib/strategy.go

func (r *Router) pickLeastConnections() (*Worker, error) {

    var best *Worker

    var minActive int64 = -1



    for _, worker := range r.workers {

        if !worker.isRoutable() {

            continue

        }



        active := worker.ActiveRequests()

        if best == nil || active < minActive {

            best = worker

            minActive = active

        }

    }



    if best == nil {

        return nil, ErrNoWorkersAvailable

    }



    return best, nil

}

Explanation:
This code selects the healthy Worker with the fewest active requests. It helps distribute requests more efficiently across Workers.

C.5 Worker gRPC Handler

File:

worker/main.py

def Handle(self, request, context):

    priority = request.priority if request.priority in (0, 1) else 1

    tier = "pro" if priority == 0 else "free"



    self.queue.push(priority, request.request_id, request.message)



    item = self.queue.pop()

    if item is None:

        context.abort(grpc.StatusCode.INTERNAL, "queue unexpectedly empty")

        return



    _, req_id, message = item



    rag_context = retrieve(message)



    reply = run_model(

        prompt=message,

        context=rag_context,

        worker_port=WORKER_PORT

    )



    depth = self.queue.size()



    return worker_pb2.Response(

        request_id=req_id,

        reply=reply,

        queue_length=depth,

    )

Explanation:
This code receives a request from the Master, adds it to the priority queue, retrieves RAG context, calls the LLM, and returns the generated answer.



C.6 RAG Retrieval

File:

worker/rag.py

def retrieve(prompt: str, top_k: int = TOP_K) -> str:

    if not DOCUMENTS:

        return ""



    terms = [t for t in prompt.lower().split() if len(t) > 2]



    scored = sorted(

        ((doc, _score(doc, terms)) for doc in DOCUMENTS),

        key=lambda x: x[1],

        reverse=True,

    )



    top_docs = [doc[:MAX_CHUNK_CHARS] for doc, s in scored[:top_k] if s > 0]



    if not top_docs:

        top_docs = [doc[:MAX_CHUNK_CHARS] for doc in DOCUMENTS[:top_k]]



    context = "\n\n".join(

        f"[{i+1}] {doc}" for i, doc in enumerate(top_docs)

    )



    return context

Explanation:
This function retrieves relevant document snippets based on the user prompt and returns them as context for the LLM.
